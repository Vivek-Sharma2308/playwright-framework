test('Amazon india Website Login Test', async ({ page, context }) => {

  await page.goto('https://www.amazon.in/', {waitUntil: 'domcontentloaded'});
  //await page.waitForTimeout(3000);
  await page.waitForFunction( ()=>document.title.length > 0 );
  // Expect a title "to contain" a substring.

  await page.getByRole('link', {name: 'Account & Lists'}).click();
  
  await expect(page).toHaveTitle('Amazon Sign-In');

  await page.fill('#ap_email_login', "vivekattri23@gmail.com");
  await page.getByRole('button', {name:/Continue/}).click();
  

  await page.fill('#ap_password', "KharchaPaani@0987");
  await page.getByRole('button', {name:/sign in/i}).click();
  
  await expect(page).toHaveTitle('Online Shopping site in India: Shop Online for Mobiles, Books, Watches, Shoes and More - Amazon.in');

  // Get the count of the items in Cart
  let cartText = await page.locator("//a[contains(@aria-label, 'items in cart')]").getAttribute('aria-label');
  console.log('Items in Cart : ', cartText);
  let cartItems: string[] = cartText?.split(' ') ?? [];
  let itemNumbers = parseInt(cartItems[0]);
  console.log("The number of items present in the carts is : ", itemNumbers);

  // Search a Mobile phone and add it to the cart

  await page.getByLabel(/Search Amazon.in/).click();
  await page.fill('#twotabsearchtextbox', 'Iphone 17 pro max');

  await page.getByRole('button', {name: 'iphone 17 pro max 256gb'}).click();

  const [newPage] = await Promise.all([
    context.waitForEvent('page'),
    page.getByRole('link', {name: 'iPhone 17 Pro 1 TB: 15.93 cm (6.3″) Display with Promotion up to 120Hz, A19 Pro Chip, Breakthrough Battery Life, Pro Fusion Camera System with Center Stage Front Camera; Deep Blue'}).first().click()
  ]);

  await newPage.waitForLoadState();
  console.log('New tab title:', await newPage.title());
  await expect(newPage).toHaveTitle('iPhone 17 Pro Smartphone');

  // Switch back to home tab
  await page.bringToFront();
  console.log('Back to home:', await page.title());

  await expect(page).toHaveTitle('Amazon.in : iphone 17 pro max 256gb');

  page.getByRole('button', {name: 'Add to cart'}).first().click();

  await page.waitForTimeout(3000);

  // Validate the count of items in cart has been increased by one
  // Get the count of the items in Cart
  cartText = await page.locator("//a[contains(@aria-label, 'items in cart')]").getAttribute('aria-label');
  console.log('Items in Cart : ', cartText);
  cartItems = cartText?.split(' ') ?? [];
  let newItemNumbers = parseInt(cartItems[0]);
  console.log("Now the number of items present in the carts is : ", newItemNumbers);

  expect((newItemNumbers - 1) == itemNumbers);

  // Remove the item added in Cart

  await page.locator("//a[contains(@aria-label, 'items in cart')]").click();

  await page.locator("//input[@value = 'Delete' and contains(@aria-label, 'Delete iPhone 17 Pro 1 TB: 15.93 cm (6.3&Prime;) Display with Promotion up to 120Hz, A19 Pro Chip')]").click();

  await page.waitForTimeout(3000);

  cartText = await page.locator("//a[contains(@aria-label, 'items in cart')]").getAttribute('aria-label');
  console.log('Items in Cart : ', cartText);

});
